## FORMAT ##
BVHFile    := Sentinel BVHNode                                    # the node is a single tree plus a sentinel
BVHNode    := EBVHBranch | EBVHLeaf | IBVHBranch | IBVHLeaf       # a node is either a leaf or a branch, implicit or explicit
IBVHBranch := IBranchID left:BVHNode right:BVHNode                # pre-order printing of the tree
EBVHBranch := EBranchID bounds:BBox left:BVHNode right:BVHNode    # pre-order printing of the tree
IBVHLeaf   := ILeafID numNode:int Tri*                            # number of Tri entries == numNode
EBVHLeaf   := ELeafID bounds:BBox numNode:int Tri*                # number of Tri entries == numNode
Tri        := p1:Point p2:Point p3:Point                          # three points of the triangle
Point      := x:float y:float z:float
BBox       := x:Range y:Range z:Range
Range      := min:float max:float                                 # note: bounds might be infinite
Tri*       :  size_t | unsigned int                               # indices for triangles, which are referenced from another file

## CONSTANTS ##
EBranchID   := 0:int
ELeafID     := 1:int
IBranchID   := 2:int
ILeafID     := 3:int
Sentinel    := 9215:int
